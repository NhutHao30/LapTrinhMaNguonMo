A. MỤC TIÊU  
− Hiểu cú pháp cơ bản và các thành phần chính của JavaScript.  
− Thao tác DOM để thay đổi nội dung và giao diện trang web.  
− Sử dụng JavaScript để gửi và nhận dữ liệu từ PHP (AJAX).  
− Kết hợp HTML, CSS, JavaScript và PHP tạo ứng dụng web tương tác.  
− Thực hành xử lý dữ liệu bất đồng bộ với Fetch API hoặc XMLHttpRequest.  
B. NỘI DUNG THỰC HÀNH  
1. Tóm tắt lý thuyết  
− Cú pháp JavaScript: biến (var, let, const), kiểu dữ liệu, toán tử, câu điều kiện, vòng lặp,

− JSON: format trao đổi dữ liệu giữa JS và PHP.

Kiểm tra email  
<!DOCTYPE  html> 
<html> 
 
<body> 
    <form onsubmit ="return validateEmail()" > 
        <input type="text" id="email" placeholder ="Nhập email" > 
        <button type="submit" >Kiểm tra </button>  
    </form> 
    <p id="msg"></p> 
    <script>  
        function  validateEmail() {  
            let email = document.getElementById( "email").value;  
            let regex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/; 
            if (regex.test(email)) {  
                document.getElementById( "msg").innerText = "Email hợp lệ" ; 
            } else { 
                document.getElementById( "msg").innerText = "Email không hợp 
lệ"; 
            } 
            return false; 
        } 
    </script>  
</body> 
 
</html>

AJAX cơ bản với PHP  
Bài tập 04:  Tạo file time.php trả về thời gian server, dùng JavaScript Fetch API hiển 
thị vào trang mỗi 5 giây.  
Hướng dẫn  
time.php  
<?php 
echo date( "H:i:s"); 
?> 
index.html  
<!DOCTYPE  html> 
<html> 
 
<body> 
    <h1 id="time"></h1> 
    <script>  
        function  getTime() {  
            fetch('time.php' ) 
                .then(res => res.text())  
                .then(data => document.getElementById( "time").innerText = 
data); 
        } 
        setInterval(getTime, 5000); 
        getTime();  
    </script>  
</body> 
 
</html> 
  Bài tập 05:  Tạo form nhập tên, gửi bằng AJAX tới hello.php, PHP trả về "Xin chào, 
[tên]" và JS hiển thị.  
Hướng dẫn  
hello.php  
<?php 
$name = $_POST[ 'name'] ?? 'Bạn'; 
echo "Xin chào, "  . htmlspecialchars($name);  
?> 
index.html  
<!DOCTYPE  html> 
<html> 
 
<body> 
    <input type="text" id="name"> 
    <button onclick="sayHello()" >Gửi</button>  
    <p id="msg"></p> 
    <script>  
        function  sayHello() {  
            let formData = new FormData();  
            formData.append( 'name', document.getElementById( "name").value);  
            fetch('hello.php' , { 
                    method: 'POST', 
                    body: formData  
                }) 
                .then(res => res.text())  
                .then(data => document.getElementById( "msg").innerText = 
data); 
        } 
    </script>  
</body> 
 
</html> 
 
Bài tập 06:  Lấy danh sách sản phẩm từ file JSON (products.json) bằng Fetch API và 
hiển thị dạng bảng.  
Hướng dẫn  
products.json  
[ 
  { "name": "Sản phẩm 1" , "price": 10000 },    { "name": "Sản phẩm 2" , "price": 20000 } 
] 
index.html  
<!DOCTYPE  html> 
<html> 
 
<body> 
    <table border="1" id="tbl"></table>  
    <script>  
        fetch('products.json' ) 
            .then(res => res.json())  
            .then(data => { 
                let html = "<tr><th>Tên</th><th>Giá</th></tr>" ; 
                data.forEach(p => { 
                    html += `<tr><td> ${p.name}</td><td> ${p.price}</td></tr>` ; 
                }); 
                document.getElementById( "tbl").innerHTML = html;  
            }); 
    </script>  
</body> 
 
</html> 
Ứng dụng kết hợp JS + PHP  
Bài tập 07:  Tạo form tìm kiếm sản phẩm: JS gửi từ khóa tới search.php, PHP trả JSON 
danh sách kết quả, JS hiển thị.  
Hướng dẫn  
search.php  
<?php 
$keyword = strtolower($_GET[ 'q'] ?? ''); 
$products = [  
    ["name" => "Iphone 15" , "price" => 30000000 ], 
    ["name" => "Samsung S24" , "price" => 25000000 ] 
]; 
$result = array_filter($products, fn($p) => strpos(strtolower($p[ 'name']), 
$keyword) !== false); 
header('Content -Type: application/json' ); 
echo json_encode(array_values($result));  
?> 
index.html  
<input type="text" id="kw" placeholder ="Nhập từ khóa" >  <button onclick="search()" >Tìm kiếm </button>  
<ul id="result" ></ul> 
<script>  
    function  search() {  
        let q = document.getElementById( "kw").value;  
        fetch('search.php?q='  + encodeURIComponent(q))  
            .then(res => res.json())  
            .then(data => { 
                document.getElementById( "result" ).innerHTML = data.map(p => 
`<li>${p.name} - ${p.price}</li>`).join(''); 
            }); 
    } 
</script>  
Bài tập 08:  Tạo ứng dụng chat đơn giản: JS gửi tin nhắn tới PHP lưu vào file, PHP trả 
lại lịch sử chat.  
Hướng dẫn  
chat.php  
<?php 
$file = 'chat.txt' ; 
if (isset($_POST[ 'msg'])) { 
    file_put_contents($file, $_POST[ 'msg'] . "\n", FILE_APPEND);  
} 
$messages = file_exists($file) ? file($file, FILE_IGNORE_NEW_LINES) : [];  
header('Content -Type: application/json' ); 
echo json_encode($messages);  
?> 
index.html  
<textarea  id="chat" rows="10" cols="30" readonly ></textarea><br>  
<input type="text" id="msg"> 
<button onclick="send()" >Gửi</button>  
<script>  
    function  loadChat() {  
        fetch('chat.php' ) 
            .then(res => res.json())  
            .then(data => { 
                document.getElementById( "chat").value = data.join( "\n"); 
            }); 
    } 
 
    function  send() {  
        let formData = new FormData();           formData.append( 'msg', document.getElementById( "msg").value);  
        fetch('chat.php' , { 
                method: 'POST', 
                body: formData  
            }) 
            .then(() => { 
                document.getElementById( "msg").value = ''; 
                loadChat();  
            }); 
    } 
    setInterval(loadChat, 2000); 
    loadChat();  
</script>

echo json_encode($data[$city] ?? [ "temp" => 0, "desc" => "Không có dữ 
liệu"]); 
?> 
index.html  
<input type="text" id="city" placeholder ="Nhập tên thành phố" > 
<button onclick="getWeather()" >Xem</button>  
<div id="weather" ></div> 
<script>  
    function  getWeather() {  
        let c = document.getElementById( "city").value;  
        fetch('weather.php?city='  + encodeURIComponent(c))  
            .then(res => res.json())  
            .then(data => { 
                document.getElementById( "weather" ).innerHTML =  
                    `<h3>${c}</h3><p>Nhiệt độ: 
${data.temp }°C</p><p> ${data.desc }</p>`;

</script>
